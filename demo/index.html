JavaScript

/* Опции: */
var options = {

	path: 'parts/img/', // Путь к каталогу с файлами изображений
	nextPostDelay: 300, // Задержка перед отображением каждой следующего изображения из группы, мс

	// Формир. разметки (разделение на колонки)
	layout: {
		// Контейнер для записей
		posts: $('<div/>', { class: 'clearfix' }),
		// Обертка для каждой записи
		postWrap:  $('<div/>', { class: '' }).css({
			float: 'left',
			width: '33.333333%'
		}),
		// HTML изображения
		postImg: $('<img>').attr('alt', 'lm-img').css({
			width: '100%'
		}),
		// Вкл. прелоадер на всех этапах
		preloader: {
			onInit: true,
			onEachItem: true,
			onButton: true
		},
	},

	// Анимация появления записей
	beforeShowItem: function(sets, curPost) {
		curPost.content.addClass('animated custom-animation-params');
	},
	afterShowItem: function(sets, curPost) {
		curPost.content.addClass('zoomIn'); // Класс animate.css 
	}

};
